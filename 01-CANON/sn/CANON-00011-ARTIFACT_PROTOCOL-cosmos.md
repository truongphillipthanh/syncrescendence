PASS QualityTierSystem:
    sutra: "MVA >> Beta >> Stable >> Canonical—few reach Canonical, most stay Stable"
    gloss:
        Four quality tiers with explicit advancement criteria. MVA is experimental
        draft. Beta is validated with small cohort. Stable is robust with diverse
        validation. Canonical is foundational with extensive peer review.
    spec:
        MVA:
            status: initial_concept | rough_draft | speculative
            validation: none_required
            iteration: rapid
        Beta:
            status: structured | coherent
            validation: small_cohort | n<10
            requirements: [structure_complete, internal_coherence, pilot_tested_3-5, feedback_incorporated]
            timeline: 1-3_months_from_MVA
        Stable:
            status: robust | comprehensive
            validation: diverse_practitioners | n≥30
            requirements: [6_months_beta, 10+_validators, quantitative_metrics, peer_review_2+, teaching_materials]
            timeline: 6-18_months_from_Beta
        Canonical:
            status: foundational | extensive_validation
            validation: supermajority_consensus | longitudinal_data
            requirements: [12_months_stable, 30+_validators, cultural_contexts, teaching_lineage, reconstructive_capacity]
            timeline: 2-5+_years_from_Stable
            critical: few_artifacts_reach | most_remain_Stable_indefinitely
end

PROC TierAdvancement:
    sutra: "Checklist validation—demonstrated competence, not time alone"
    gloss:
        Each tier transition has explicit requirements: structure, validation,
        peer review, documentation. Artifacts can move DOWN tiers if evidence
        weakens or alternatives emerge.
    spec:
        MVA_to_Beta:
            requirements:
                - complete_structure
                - internal_coherence
                - integration_points_identified
                - pilot_tested_3-5
                - feedback_incorporated
                - limitations_stated
                - founder_approval
        Beta_to_Stable:
            requirements:
                - 6_months_beta_minimum
                - 10+_practitioners_validated
                - diverse_cognitive_profiles
                - quantitative_metrics_if_applicable
                - peer_review_2+_Scale_3-Macro
                - community_consensus_emerging
                - maintenance_protocol_established
        Stable_to_Canonical:
            requirements:
                - 12_months_stable_minimum
                - 30+_validators
                - multiple_cultural_contexts
                - longitudinal_12+_months
                - framework_integration_complete
                - teaching_lineage_demonstrated
                - reconstructive_capacity_verified
        degradation_possible:
            Canonical_to_Stable: empirical_falsification | better_alternatives | consensus_shifts
            Stable_to_Beta: validation_failure_new_populations | outcome_data_weakens
            Beta_to_MVA: conceptual_errors | complete_redesign_required
            MVA_to_DEPRECATED: experiment_failed | community_abandonment
end

PROC MaintenanceEvolution:
    sutra: "Semantic versioning—MAJOR breaking, MINOR additions, PATCH corrections"
    gloss:
        Versioning follows semantic format: MAJOR.MINOR.PATCH. Maintenance
        rhythms: continuous (typos), quarterly (feedback), annual (comprehensive),
        multi-year (canonical review). Deprecation preserves historical context.
    spec:
        versioning:
            MAJOR: fundamental_reconceptualization | breaking_changes | migration_guide_required
            MINOR: significant_additions | enhanced_clarity | backward_compatible
            PATCH: typo_corrections | clarifications | formatting
        maintenance_rhythms:
            continuous: typo_immediate | links_immediate | clarifications_7_days | critical_24_hours
            quarterly: feedback_review | minor_version_consideration | integration_check | measurement_update
            annual: comprehensive_review | major_version_consideration | tier_assessment | archive_candidacy
            multi_year: canonical_review | historical_analysis | community_retrospective | framework_wide_audit
        deprecation_protocol:
            1: identify_candidate | community_discussion | replacement_verification | impact_analysis
            2: announce_timeline_6-12_months | mark_DEPRECATED | migration_path | transition_support
            3: move_to_archive | historical_context | redirect_references | celebrate_learning
            never_delete: all_preserved_for_historical_understanding
end
