PROC TwitterThreadProduction:
    sutra: "90-120 min—concept crystallization, hook optimization, thread construction, clarity pass"
    gloss:
        Complete workflow for Tier 2 Short-Form Twitter thread production.
        Six phases from concept to distribution. Hook tweet determines 80%
        of performance—optimize ruthlessly.
    spec:
        Phase_1_Concept_Crystallization:
            time: 15-20_min
            source: synthesis_session_output | IIC_insight
            questions:
                - core_claim_or_question_driving_thread
                - why_matters_to_audience_now
                - unique_Syncrescendent_perspective
            AI_prompt_Claude:
                input: topic | thesis | target_audience | dimensional_frameworks
                output: hook_tweet | 3-5_supporting_points | synthesis_CTA
        Phase_2_Hook_Optimization:
            time: 15-20_min
            principle: hook_determines_80%_performance
            AI_prompt_ChatGPT:
                input: draft_hook | thread_thesis | audience | goal
                request: 5-7_variations | intellectual_curiosity | immediate_value | <280_chars
            selection: genuine_curiosity | clear_value | brand_alignment | no_hype
        Phase_3_Thread_Construction:
            time: 30-40_min
            structure:
                Tweet_1_Hook: compelling_question_or_claim
                Tweet_2_Context: why_matters | stakes
                Tweet_3-4_Framework: dimensional_analysis | Syncrescendent_perspective
                Tweet_5-8_Argumentation: evidence | reasoning | examples
                Tweet_9-10_Synthesis: integration | implications
                Tweet_11_CTA: newsletter | essay_link | resource
            protocol:
                - draft_single_document_first
                - one_idea_per_tweet
                - visual_breaks | line_breaks_lists
                - maintain_authoritative_register
        Phase_4_Clarity_Pass:
            time: 15-20_min
            tools: Hemingway_Editor | similar
            targets:
                grade_level: 8-10
                sentence_structure: vary_length
                passive_voice: minimize
                adverbs: reduce
            human_check: intellectual_rigor | peer_pride | positioning | unique_synthesis
        Phase_5_Visual_Formatting:
            time: 10-15_min
            actions:
                - line_breaks_for_readability
                - bullet_points_where_appropriate
                - emojis_sparingly_if_authentic
                - check_character_counts
                - ensure_logical_flow
        Phase_6_Publishing:
            time: 10-15_min
            optimal_timing:
                weekday_mornings: 7-9_AM_Pacific
                weekday_evenings: 5-7_PM_Pacific
                avoid: late_night | early_morning | weekends
            tactics:
                - use_native_composer | better_algorithm
                - publish_entire_thread_at_once
                - pin_if_strategically_important
                - cross_post_to_newsletter
            engagement:
                - monitor_first_2_hours_actively
                - respond_substantive_only
                - ignore_trolls | zero_energy
                - quote_tweet_strong_responses
end
