    spec:
        states:
            peak_80_100:
                characteristics: high_vitality | sharp_focus | enthusiastic | connected
                protocol: ambitious_development | multi_chain | complex_problems | 45-60min_sessions
                warning: "Peak creates illusion this is normal—don't commit beyond capacity"
            baseline_60_80:
                characteristics: adequate_energy | good_enough_focus | stable | grounded
                protocol: standard_practice | single_chain | routine_skill | 15-30min_sessions
                principle: "Most protocols should target baseline, not peak"
            depletion_40_60:
                characteristics: noticeable_fatigue | difficulty_focusing | irritable | disconnected
                protocol: Scale_1_Micro_ONLY | seven_pulses_minimum | zero_guilt | rest_priority

            crisis_sub_40:

                protocol: ALL_practice_suspended | survival_mode | professional_support_if_available
                recovery: 48-72hr_rest_minimum | gradual_return_only
end
