PASS RhetoricalFunctions:
    sutra: "10 purposes: Explain, Persuade, Demonstrate, Synthesize, Question, Analyze, Inspire, Contextualize, Critique, Connect"
    spec:
        explain:
            job: clarify_complex >> systematic_breakdown + reconstruction
            structure: phenomenon → confusion → parts → relationships → whole → examples
            anti_patterns: [assume_knowledge, over_complicate, miss_confusion_point]
        persuade:
            job: shift_perspective >> reasoned_argument
            structure: acknowledge → stakes → evidence → counterarguments → implications → invite
            anti_patterns: [emotion_only, ignore_counters, false_dichotomies]
        demonstrate:
            job: show_implementation >> worked_examples
            structure: context → steps → decisions → errors → results → enable_replication
            anti_patterns: [demo_without_explain, cherry_pick, hide_complexity]
        synthesize:
            job: integrate_perspectives >> coherent_understanding
            structure: target → components → tensions → pattern → emergence → validate
            anti_patterns: [false_equivalence, force_integration, lose_distinct_value]
        question:
            job: provoke_thinking >> reveal_assumptions
            structure: surface → worth_questioning → alternatives → implications → tension → invite
            anti_patterns: [fake_questions, no_guidance, contrarianism_as_identity]
        analyze:
            job: examine_systematically >> reveal_structure
            structure: target → dimensions → examine → patterns → implications → applications
            anti_patterns: [paralysis, miss_forest, impose_frameworks]
        inspire:
            job: energize_action >> reveal_possibility + model_commitment
            structure: acknowledge → reveal → path → model → ground → enable
            anti_patterns: [motivation_porn, no_instruction, unrealistic]
        contextualize:
            job: situate_in_landscape >> perspective
            structure: focal → terrain → evolution → position → trajectory → navigate
            anti_patterns: [context_without_focus, false_lineages, determinism]
        critique:
            job: identify_limitations >> enable_improvement
            structure: acknowledge_value → limitation → matters → direction → generative → alternatives
            anti_patterns: [no_alternatives, tear_down_for_status, perfectionism_weapon]
        connect:
            job: build_relationship >> authentic_sharing
            structure: personal_moment → vulnerability → themes → reciprocal → fabric → boundaries
            anti_patterns: [performative_authenticity, oversharing, missing_substance]
end

    sutra: "Empirical data, Conceptual logic, Personal narrative, Historical precedent, Analogical reasoning"
    spec:
        empirical:
            definition: measurable_observations | statistical | experimental
            quality: primary_preferred | recent | representative | honest_uncertainty
            anti_patterns: [cherry_picking, misleading_viz, false_precision]
        conceptual:
            definition: reasoning_from_principles | thought_experiments

            anti_patterns: [circular, hidden_premises, leaps, complexity_obfuscation]
        personal:
