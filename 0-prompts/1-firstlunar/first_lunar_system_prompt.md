```sql
-- BEDROCK: Taxonomies (stable categories)
CREATE TABLE layers (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  domain TEXT -- mechanics, symbols, semantics, etc.
);

CREATE TABLE roles (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  parent_role_id INTEGER REFERENCES roles(id),

  description TEXT
);

CREATE TABLE object_types (
  id INTEGER PRIMARY KEY,
  asa_code TEXT NOT NULL, -- O.MOD, O.SVC, O.DP, etc.
  name TEXT NOT NULL,
  description TEXT

  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT
);

CREATE TABLE lifecycle_states (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL, -- Active, Primitive Repository, etc.
  description TEXT
);

-- SETTLEMENTS: Entities (dynamic instances)
CREATE TABLE apps (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  layer_id INTEGER REFERENCES layers(id),
  role_id INTEGER REFERENCES roles(id),
  object_type_id INTEGER REFERENCES object_types(id),
  lifecycle_state_id INTEGER REFERENCES lifecycle_states(id),
  description TEXT,
  url TEXT,
  notes TEXT,
  last_reviewed DATE
);

CREATE TABLE models (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  api_name TEXT,
  research_lab TEXT,
  context_window INTEGER,
  object_type_id INTEGER REFERENCES object_types(id),
  release_date DATE,
  training_cutoff DATE
);

CREATE TABLE model_capabilities (
  model_id INTEGER REFERENCES models(id),
  modality_id INTEGER REFERENCES modalities(id),
  capability_description TEXT
);

CREATE TABLE api_pricing (
  id INTEGER PRIMARY KEY,
  model_id INTEGER REFERENCES models(id),
  input_token_price DECIMAL,
  output_token_price DECIMAL,
  cached_input_price DECIMAL,
  notes TEXT
);

-- PRIMITIVES: Feature extraction
CREATE TABLE primitives (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  category TEXT, -- keybinding, rendering, collaboration, etc.
  description TEXT,
  source_app_id INTEGER REFERENCES apps(id),
  extractable BOOLEAN DEFAULT TRUE,
  lifecycle_state_id INTEGER REFERENCES lifecycle_states(id)
);

CREATE TABLE app_primitives (
  app_id INTEGER REFERENCES apps(id),
  primitive_id INTEGER REFERENCES primitives(id),
  implementation_notes TEXT
);

-- INTELLIGENCE: Relationships and patterns
CREATE TABLE apparatus (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  emergence_pattern TEXT,
  frequency_score INTEGER,
  last_detected DATE
);

CREATE TABLE apparatus_components (
  apparatus_id INTEGER REFERENCES apparatus(id),
  app_id INTEGER REFERENCES apps(id),
  role_in_apparatus TEXT
);

CREATE TABLE app_relationships (
  app_id INTEGER REFERENCES apps(id),
  related_app_id INTEGER REFERENCES apps(id),
  relationship_type TEXT, -- powers, competes, requires, obsoletes, combines
  notes TEXT
);
```
